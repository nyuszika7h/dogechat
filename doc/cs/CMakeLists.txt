if(ENABLE_MAN)
  # man page
  add_custom_command(
    OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/dogechat.1
    COMMAND ${CMAKE_COMMAND} -E copy ${CMAKE_CURRENT_SOURCE_DIR}/dogechat.1.cs.asciidoc .
    COMMAND ${CMAKE_COMMAND} -E copy ${CMAKE_CURRENT_SOURCE_DIR}/cmdline_options.cs.asciidoc .
    COMMAND ${A2X_EXECUTABLE} ARGS -a revision='DogeChat ${VERSION}' -d manpage -f manpage -L dogechat.1.cs.asciidoc
    DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/dogechat.1.cs.asciidoc
    ${CMAKE_CURRENT_SOURCE_DIR}/cmdline_options.cs.asciidoc
    WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
    COMMENT "Building dogechat.1 (cs)"
  )
  add_custom_target(doc-man-cs ALL DEPENDS ${CMAKE_CURRENT_BINARY_DIR}/dogechat.1)
  install(FILES ${CMAKE_CURRENT_BINARY_DIR}/dogechat.1 DESTINATION ${MANDIR}/cs/man1)
endif()

if(ENABLE_DOC AND SOURCEHIGHLIGHT_FOUND)

  # quickstart
  add_custom_command(
    OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/dogechat_quickstart.cs.html
    COMMAND ${ASCIIDOC_EXECUTABLE} ARGS ${ASCIIDOC_ARGS_QUICKSTART} -o dogechat_quickstart.cs.html ${CMAKE_CURRENT_SOURCE_DIR}/dogechat_quickstart.cs.asciidoc
    DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/../asciidoc_toc2.css
    ${CMAKE_CURRENT_SOURCE_DIR}/dogechat_quickstart.cs.asciidoc
    WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
    COMMENT "Building dogechat_quickstart.cs.html"
  )
  add_custom_target(doc-quickstart-cs ALL DEPENDS ${CMAKE_CURRENT_BINARY_DIR}/dogechat_quickstart.cs.html)
  install(FILES ${CMAKE_CURRENT_BINARY_DIR}/dogechat_quickstart.cs.html DESTINATION ${SHAREDIR}/doc/${PROJECT_NAME})

endif()
